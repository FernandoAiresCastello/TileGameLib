

	LIST OF TGML COMMANDS


	* Note: Identifiers between parentheses indicate values from the stack


-------------------------------------------------------------------------------
NOP
	No operation

-------------------------------------------------------------------------------
EXIT
	Exit the interpreter and close application

-------------------------------------------------------------------------------
JMP label
	Jump to label

-------------------------------------------------------------------------------
JZ label
	Jump to label if top of stack = 0

-------------------------------------------------------------------------------
JNZ label
	Jump to label if top of stack != 0

-------------------------------------------------------------------------------
CALL label
	Call subroutine

-------------------------------------------------------------------------------
CLZ label
	Call subroutine if top of stack = 0

-------------------------------------------------------------------------------
CLNZ label
	Call subroutine if top of stack != 0

-------------------------------------------------------------------------------
RET
	Return from subroutine

-------------------------------------------------------------------------------
SLEEP (ms)
	Pause interpreter for ms milliseconds

-------------------------------------------------------------------------------
PUSH value
	Insert value at top of stack

-------------------------------------------------------------------------------
STORE variable
	Store value at top of stack into variable

-------------------------------------------------------------------------------
LOAD variable
	Load value from variable and insert at top of stack

-------------------------------------------------------------------------------
MATH.INC (number)
	Increment number by 1

-------------------------------------------------------------------------------
MATH.DEC (number)
	Decrement number by 1

-------------------------------------------------------------------------------
MATH.ADD (number1, number2)
	Get result of number1 + number2

-------------------------------------------------------------------------------
MATH.SUBTR (number1, number2)
	Get result of number1 - number2

-------------------------------------------------------------------------------
MATH.COMPARE (number1, number2)
	Compare two numbers (get result of number1 - number2)

-------------------------------------------------------------------------------
MATH.MULT (number1, number2)
	Get result of number1 * number2

-------------------------------------------------------------------------------
MATH.DIVIDE (number1, number2)
	Get result of number1 / number2

-------------------------------------------------------------------------------
MATH.MODULO (number1, number2)
	Get result of number1 % number2

-------------------------------------------------------------------------------
MATH.SQRT (number)
	Get square root of number

-------------------------------------------------------------------------------
MATH.POWER (number, exponent)
	Get power of number

-------------------------------------------------------------------------------
MAP.LOAD (path)
	Load map from file

-------------------------------------------------------------------------------
WINDOW.OPEN (cols, rows)
	Open graphics window

-------------------------------------------------------------------------------
WINDOW.REFRESH
	Refresh window

-------------------------------------------------------------------------------
WINDOW.PRINT (string)
	Print string in window

	TGML COMMANDS
-------------------------------------------------------------------------------
WINDOW.KEY.IS_PRESSED (key)
	Get 1 if key is pressed, 0 otherwise

OBJECT.FIND (id)
	Find object with id and get layer, x, y

OBJECT.ID.SET (layer, x, y, id)
OBJECT.ID.GET (layer, x, y)
	Set or get object id

OBJECT.DATA.SET (layer, x, y, data)
OBJECT.DATA.GET (layer, x, y)
	Set or get object data

OBJECT.TILE.IX.SET (layer, x, y, frame, index)
OBJECT.TILE.IX.GET (layer, x, y, frame)
	Set or get object tile index
	
OBJECT.TILE.FG.SET (layer, x, y, frame, forecolor)
OBJECT.TILE.FG.GET (layer, x, y, frame)
	Set or get object tile foreground color

OBJECT.TILE.BG.SET (layer, x, y, frame, backcolor)
OBJECT.TILE.BG.GET (layer, x, y, frame)
	Set or get object tile background color

OBJECT.MOVE (src_layer, src_x, src_y, dest_layer, dest_x, dest_y)
	Move object from source to destination

OBJECT.COPY (src_layer, src_x, src_y, dest_layer, dest_x, dest_y)
	Copy object from source and paste at destination

OBJECT.SWAP (layer1, x1, y1, layer2, x2, y2)
	Swap location of two objects

OBJECT.DELETE (layer, x, y)
	Delete object

	TGML COMMANDS
-------------------------------------------------------------------------------
WINDOW.KEY.IS_PRESSED (key)
	Get 1 if key is pressed, 0 otherwise

WINDOW.PALETTE.SET (index, rgb)
WINDOW.PALETTE.GET (index)
	Set or get window palette color

WINDOW.TILESET.SET (index, pattern)
WINDOW.TILESET.GET (index)
	Set or get window tileset pattern

MAP.NAME.SET (name)
MAP.NAME.GET
	Set or get map name

MAP.BACKCOLOR.SET (color)
MAP.BACKCOLOR.GET
	Set or get map background color

MAP.PALETTE.SET (index, rgb)
MAP.PALETTE.GET (index)
	Set or get map palette color

MAP.TILESET.SET (index, pattern)
MAP.TILESET.GET (index)
	Set or get map tileset pattern

OBJECT.FIND (id)
	Find object with id and get layer, x, y

OBJECT.ID.SET (layer, x, y, id)
OBJECT.ID.GET (layer, x, y)
	Set or get object id

OBJECT.DATA.SET (layer, x, y, data)
OBJECT.DATA.GET (layer, x, y)
	Set or get object data

OBJECT.TILE.INDEX.SET (layer, x, y, frame, index)
OBJECT.TILE.INDEX.GET (layer, x, y, frame)
	Set or get object tile index
	
OBJECT.TILE.FORECOLOR.SET (layer, x, y, frame, forecolor)
OBJECT.TILE.FORECOLOR.GET (layer, x, y, frame)
	Set or get object tile foreground color

OBJECT.TILE.BACKCOLOR.SET (layer, x, y, frame, backcolor)
OBJECT.TILE.BACKCOLOR.GET (layer, x, y, frame)
	Set or get object tile background color

OBJECT.MOVE (src_layer, src_x, src_y, dest_layer, dest_x, dest_y)
	Move object from source to destination

OBJECT.COPY (src_layer, src_x, src_y, dest_layer, dest_x, dest_y)
	Copy object from source and paste at destination

OBJECT.SWAP (layer1, x1, y1, layer2, x2, y2)
	Swap location of two objects

OBJECT.DELETE (layer, x, y)
	Delete object
